                      # Kafka

                      # 什么是 Kafka

                      一个分布式流式处理平台，它的两大应用场景是：

                      1. 消息队列：建立实时流数据管道，以可靠地在系统或应用程序之间获取数据；
                      2. 数据处理：构建实时的流数据处理程序来转换或处理数据流。

                      Kafka 相比其他消息队列主要的优势如下：

                      1. 性能优越，Kafka 基于 Scala 和 Java 语言开发，设计中大量使用了批量处理和异步的思想，最高可以每秒处理千万级别的消息
                      2. Kafka 与周边生态系统的兼容性是最好的，尤其在大数据和流计算领域

                      作为消息队列，Kafka 是一种发布-订阅模型，使用主题(Topic)作为消息通信载体，类似广播模式，发布者发布一条消息，通过主题传递给所有订阅者，而发布前没有订阅的用户是接收不到的

                      # Kafka - 实现发布和订阅

                      使用前需要下载安装 zookeeper、Kafka，记录安装步骤：https://blog.csdn.net/qq_35461948/article/details/117772904

                      在 Spring boot 中添加依赖：

                      <dependency>
    <groupId>org.springframework.kafka</groupId>
    <artifactId>spring-kafka</artifactId>
</dependency>

                      相关配置：

                      spring.kafka.bootstrap-servers=localhost:9092
spring.kafka.consumer.group-id=myGroup

                      接着创建向 Kafka 发送消息的生产者：

                      package com.dolpvv.ssmp.service;

import com.dolpvv.ssmp.entity.FMData;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.kafka.support.SendResult;
import org.springframework.stereotype.Service;
import org.springframework.util.concurrent.ListenableFuture;
import org.springframework.util.concurrent.ListenableFutureCallback;

import java.util.List;
import java.util.stream.Collectors;


@Slf4j
@Service
public class KafkaProducerService {

    @Autowired
    private KafkaTemplate<String, Object> kafkaTemplate;

    public void sendMessage(String topic, List<FMData> obj) {

        // 表示发送 Kafka 消息的异步操作的 Future 对象
        // 当使用 KafkaTemplate 发送消息时，返回 Future 对象，可以用于异步地获取发送结果
        ListenableFuture<SendResult<String, Object>> future = 
            kafkaTemplate.send(topic, obj);

        // addCallback 允许向 Future 对象添加回调函数，以便在异步操作完成时执行相应的逻辑
        future.addCallback(
            new ListenableFutureCallback<SendResult<String, Object>>() {
                @Override
                public void onFailure(Throwable ex) {
                    log.info("Fail to send {} message:{}, reason:{}",
                            topic,
                            obj.toString(),
                            ex.getMessage()
                    );
                }

                @Override
                public void onSuccess(SendResult<String, Object> result) {
                    log.info("Successfully send message to topic:{} -> {}",
                            topic,
                            result.getProducerRecord().value().toString()
                    );
                    String op = topic.equals("ADD") ? "新增" : "清除";
                    log.info("接收到 {} {} 条告警信息，流水号为：{}",
                            op,
                            obj.size(),
                            obj.stream()
                            .map(FMData::getCsn)
                            .map(Object::toString)
                            .collect(Collectors.joining(", "))
                    );
                }
            }
        );
    }
}

                      创建接收消息的消费者：

                      package com.dolpvv.ssmp.service;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.opencsv.CSVWriter;
import lombok.extern.slf4j.Slf4j;
import org.apache.kafka.clients.consumer.ConsumerRecord;
import org.springframework.kafka.annotation.KafkaListener;
import org.springframework.stereotype.Service;

import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.StreamSupport;

@Slf4j
@Service
public class KafkaConsumerService {

    // 监听两个主题
    @KafkaListener(topics = {"ADD", "CLEAR"})
    public void receiveAddMessage(ConsumerRecord<String, String> consumerRecord) 
        throws IOException {

        log.info("Receive message from topic:{} partition:{} -> {}",
                consumerRecord.topic(),
                consumerRecord.partition(),
                consumerRecord.value()
        );

        // 向本地 csv 文件添加
        if (consumerRecord.topic().equals("ADD")) {
            String csv = "D:/FMDataQuery/output/fmdata.csv";
            boolean append = new File(csv).exists();
            CSVWriter csvWriter = new CSVWriter(new FileWriter(csv, append));

            ObjectMapper objectMapper = new ObjectMapper();
            JsonNode jsonNode = objectMapper.readTree(consumerRecord.value());

            // 第一次创建文件时添加表头
            if (!append) {
                List<String> headers = new ArrayList<>();
                Iterator<String> fieldNames = jsonNode.get(0).fieldNames();
                while (fieldNames.hasNext()) {
                    headers.add(fieldNames.next());
                }
                csvWriter.writeNext(headers.toArray(new String[0]));
            }

            List<String[]> rows = StreamSupport
                .stream(jsonNode.spliterator(), false)
                .map(node -> StreamSupport.stream(node.spliterator(), false)
                .map(JsonNode::asText)
                .toArray(String[]::new))
                .collect(Collectors.toList());

            csvWriter.writeAll(rows);
            csvWriter.close();
        }
    }
}
